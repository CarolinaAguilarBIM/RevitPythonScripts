# -*- coding: utf-8 -*-
import clr
clr.AddReference('RevitAPI')
clr.AddReference('RevitAPIUI')

from Autodesk.Revit.DB import *
from Autodesk.Revit.UI import *

# Get current Revit document
doc = __revit__.ActiveUIDocument.Document

# Collect all view filters (ParameterFilterElement)
filters = FilteredElementCollector(doc).OfClass(ParameterFilterElement).ToElements()

# Print all filter names
print("ðŸ“‹ View Filters in the current project:\n")
for f in filters:
    print("- {0}".format(f.Name))

print("\nâœ… Done! {0} filters found.".format(len(filters)))
